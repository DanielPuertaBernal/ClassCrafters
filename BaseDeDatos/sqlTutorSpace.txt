-- Tabla País
CREATE TABLE Pais (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

-- Tabla Departamento
CREATE TABLE Departamento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    pais_id INT,
    FOREIGN KEY (pais_id) REFERENCES Pais(id)
);

-- Tabla Ciudad
CREATE TABLE Ciudad (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    departamento_id INT,
    FOREIGN KEY (departamento_id) REFERENCES Departamento(id)
);

-- Tabla TipoDocumento
CREATE TABLE TipoDocumento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

-- Tabla Institucion
CREATE TABLE Institucion (
    id SERIAL PRIMARY KEY,
    tipo_documento_id INT,
    numero_documento VARCHAR(50) NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    FOREIGN KEY (tipo_documento_id) REFERENCES TipoDocumento(id)
);

-- Tabla TipoSucursal
CREATE TABLE TipoSucursal (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

-- Tabla SucursalInstitucion
CREATE TABLE SucursalInstitucion (
    id SERIAL PRIMARY KEY,
    nombre_sucursal VARCHAR(100) NOT NULL,
    tipo_sucursal_id INT,
    ciudad_id INT,
    direccion VARCHAR(255) NOT NULL,
    contacto VARCHAR(100),
    institucion_id INT,
    FOREIGN KEY (tipo_sucursal_id) REFERENCES TipoSucursal(id),
    FOREIGN KEY (ciudad_id) REFERENCES Ciudad(id),
    FOREIGN KEY (institucion_id) REFERENCES Institucion(id)
);

-- Tabla LugarTutoria
CREATE TABLE LugarTutoria (
    id SERIAL PRIMARY KEY,
    sucursal_institucion_id INT,
    nombre VARCHAR(100) NOT NULL,
    cantidad_estudiantes INT NOT NULL,
    FOREIGN KEY (sucursal_institucion_id) REFERENCES SucursalInstitucion(id)
);
